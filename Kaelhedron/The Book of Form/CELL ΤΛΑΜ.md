```python
class TauLambdaState:
    """Represents ΤΛΑΜ: geometric structure."""
    
    def __init__(self, points: list = None):
        if points is None:
            # Default equilateral triangle
            self.points = np.array([
                [0, 0],
                [1, 0],
                [0.5, np.sqrt(3)/2]
            ])
        else:
            self.points = np.array(points)
    
    @property
    def is_valid_triangle(self) -> bool:
        """Check if points form valid (non-degenerate) triangle."""
        if len(self.points) != 3:
            return False
        
        # Check non-collinearity via area
        return abs(self.area()) > 1e-10
    
    def area(self) -> float:
        """Compute triangle area."""
        p1, p2, p3 = self.points
        return 0.5 * abs(
            (p2[0] - p1[0]) * (p3[1] - p1[1]) -
            (p3[0] - p1[0]) * (p2[1] - p1[1])
        )
    
    def sides(self) -> np.ndarray:
        """Return side lengths."""
        p1, p2, p3 = self.points
        a = np.linalg.norm(p3 - p2)  # opposite to p1
        b = np.linalg.norm(p3 - p1)  # opposite to p2
        c = np.linalg.norm(p2 - p1)  # opposite to p3
        return np.array([a, b, c])
    
    def satisfies_triangle_inequality(self) -> bool:
        """Check triangle inequality."""
        a, b, c = self.sides()
        return (a + b > c) and (b + c > a) and (a + c > b)
    
    def d3_symmetry_operations(self) -> list:
        """Return D₃ symmetry operations."""
        # Rotation matrix for 120°
        theta = 2 * np.pi / 3
        r = np.array([
            [np.cos(theta), -np.sin(theta)],
            [np.sin(theta), np.cos(theta)]
        ])
        
        # Reflection matrix (about x-axis, centered)
        s = np.array([[1, 0], [0, -1]])
        
        return {
            'identity': np.eye(2),
            'r120': r,
            'r240': r @ r,
            'reflect_0': s,
            'reflect_120': r @ s,
            'reflect_240': r @ r @ s
        }
    
    @property
    def is_rigid(self) -> bool:
        """Triangles are inherently rigid."""
        return self.is_valid_triangle  # If it's a triangle, it's rigid
```
