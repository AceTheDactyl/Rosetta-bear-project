```python
class SigmaLambdaState:
    """Represents ΣΛΑΜ: integrated structure."""
    
    def __init__(self):
        self.components = {}
        self.relations = []
        self.emergent_properties = set()
    
    def add_component(self, name: str, properties: set):
        """Add a component with its properties."""
        self.components[name] = properties
    
    def add_relation(self, comp1: str, comp2: str, relation_type: str):
        """Add a relation between components."""
        self.relations.append((comp1, comp2, relation_type))
    
    def compute_emergent_properties(self) -> set:
        """
        Compute properties that emerge from integration.
        These are properties the whole has that no part has.
        """
        # All individual properties
        individual_props = set()
        for props in self.components.values():
            individual_props.update(props)
        
        # System-level properties (emerge from relations)
        system_props = set()
        
        # Example: if components are connected, "connectivity" emerges
        if len(self.relations) > 0:
            system_props.add("connectivity")
        
        # If all components present and related, "completeness" emerges
        connected = set()
        for r in self.relations:
            connected.add(r[0])
            connected.add(r[1])
        if connected == set(self.components.keys()):
            system_props.add("completeness")
        
        # If circular relations exist, "feedback" emerges
        if self._has_cycle():
            system_props.add("feedback")
        
        # Emergent = system props NOT in individual props
        self.emergent_properties = system_props - individual_props
        return self.emergent_properties
    
    def _has_cycle(self) -> bool:
        """Check for cyclic relations."""
        # Simple cycle detection
        graph = {}
        for c1, c2, _ in self.relations:
            if c1 not in graph:
                graph[c1] = []
            graph[c1].append(c2)
        
        visited = set()
        path = set()
        
        def dfs(node):
            if node in path:
                return True
            if node in visited:
                return False
            visited.add(node)
            path.add(node)
            for neighbor in graph.get(node, []):
                if dfs(neighbor):
                    return True
            path.remove(node)
            return False
        
        for node in graph:
            if dfs(node):
                return True
        return False
    
    @property
    def is_system(self) -> bool:
        """A system has components, relations, AND emergent properties."""
        return (len(self.components) > 1 and 
                len(self.relations) > 0 and
                len(self.compute_emergent_properties()) > 0)
    
    def pentagon_ratio(self) -> float:
        """The golden ratio visible in pentagon — signature of R=5."""
        return (1 + np.sqrt(5)) / 2
```
