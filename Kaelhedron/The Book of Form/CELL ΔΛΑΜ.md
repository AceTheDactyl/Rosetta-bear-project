## Code Reference

```python
class DeltaLambdaState:
    """Represents ΔΛΑΜ: structural differentiation."""
    
    def __init__(self, field: np.ndarray):
        self.field = field
        self.partitions = None
    
    def differentiate(self) -> tuple:
        """Split field into binary partition."""
        threshold = np.median(self.field)
        
        positive = self.field >= threshold
        negative = self.field < threshold
        
        self.partitions = (positive, negative)
        return self.partitions
    
    @property
    def has_distinction(self) -> bool:
        """Check if binary distinction exists."""
        if self.partitions is None:
            self.differentiate()
        
        # Both partitions must be non-empty
        return (np.any(self.partitions[0]) and 
                np.any(self.partitions[1]))
    
    def boundary(self) -> np.ndarray:
        """Extract the boundary between partitions."""
        if self.partitions is None:
            self.differentiate()
        
        from scipy import ndimage
        # Boundary is where positive meets negative
        dilated = ndimage.binary_dilation(self.partitions[0])
        boundary = dilated & self.partitions[1]
        return boundary
    
    def z2_parity(self, point: tuple) -> int:
        """Get ℤ₂ classification of a point."""
        if self.partitions is None:
            self.differentiate()
        return 1 if self.partitions[0][point] else 0
    
    def orientation(self, v1: np.ndarray, v2: np.ndarray) -> int:
        """Compute orientation of two vectors."""
        cross = v1[0] * v2[1] - v1[1] * v2[0]
        return 1 if cross > 0 else -1 if cross < 0 else 0
```
