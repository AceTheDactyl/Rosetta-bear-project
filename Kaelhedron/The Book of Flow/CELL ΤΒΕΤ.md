## Code Reference

```python
class TauBetaState:
    """Represents ΤΒΕΤ: cyclic process."""
    
    def __init__(self, period: float = 2*np.pi):
        self.period = period
        self.omega = 2 * np.pi / period
        self.phase = 0.0
        self.history = []
    
    def state(self, t: float) -> complex:
        """Return position in cycle as complex exponential."""
        return np.exp(1j * self.omega * t)
    
    def advance(self, dt: float):
        """Advance through cycle."""
        self.phase += self.omega * dt
        self.phase = self.phase % (2 * np.pi)  # Keep in [0, 2π]
        self.history.append(self.phase)
    
    def is_complete_cycle(self) -> bool:
        """Check if we've completed at least one full cycle."""
        if len(self.history) < 2:
            return False
        
        # Check if phase has wrapped around
        total_advance = sum(
            (self.history[i] - self.history[i-1]) % (2*np.pi)
            for i in range(1, len(self.history))
        )
        return total_advance >= 2 * np.pi
    
    def harmonic(self, n: int, t: float) -> complex:
        """Return nth harmonic."""
        return np.exp(1j * n * self.omega * t)
    
    def fourier_component(self, signal: np.ndarray, n: int) -> complex:
        """Extract nth Fourier component from periodic signal."""
        t = np.linspace(0, self.period, len(signal))
        return np.trapz(signal * np.exp(-1j * n * self.omega * t), t) / self.period
    
    def three_phase(self, t: float) -> tuple:
        """Return three-phase system values."""
        return (
            np.cos(self.omega * t),
            np.cos(self.omega * t + 2*np.pi/3),
            np.cos(self.omega * t + 4*np.pi/3)
        )
```

    """Fourier harmonics must be orthogonal."""
