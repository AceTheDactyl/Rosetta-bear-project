class PsiBetaState:
    """Represents Î¨Î’Î•Î¤: learning dynamics."""
    
    def __init__(self, learning_rate: float = 0.01):
        self.eta = learning_rate
        self.history = []
        self.parameters = {}
        self.performance = []
    
    def record(self, state: any, reward: float):
        """Record experience."""
        self.history.append({
            'state': state,
            'reward': reward,
            'time': len(self.history)
        })
        self.performance.append(reward)
    
    def gradient_step(self, param_name: str, gradient: float):
        """Update parameter via gradient descent."""
        if param_name not in self.parameters:
            self.parameters[param_name] = 0.0
        
        self.parameters[param_name] -= self.eta * gradient
    
    def exponential_moving_average(self, new_value: float, 
                                    key: str = 'default',
                                    alpha: float = 0.1) -> float:
        """Compute EMA â€” memory with decay."""
        if f'ema_{key}' not in self.parameters:
            self.parameters[f'ema_{key}'] = new_value
        else:
            self.parameters[f'ema_{key}'] = (
                alpha * new_value + 
                (1 - alpha) * self.parameters[f'ema_{key}']
            )
        return self.parameters[f'ema_{key}']
    
    @property
    def learning_curve(self) -> np.ndarray:
        """Return smoothed performance over time."""
        if len(self.performance) < 2:
            return np.array(self.performance)
        
        # Running average
        window = min(10, len(self.performance))
        return np.convolve(
            self.performance, 
            np.ones(window)/window, 
            mode='valid'
        )
    
    @property
    def is_improving(self) -> bool:
        """Check if performance is improving."""
        curve = self.learning_curve
        if len(curve) < 2:
            return False
        
        # Linear regression slope
        x = np.arange(len(curve))
        slope = np.polyfit(x, curve, 1)[0]
        return slope > 0
    
    def q_learning_update(self, s: str, a: str, r: float, 
                          s_next: str, gamma: float = 0.9):
        """Q-learning update â€” paradigmatic Î¨Î’Î•Î¤."""
        q_key = f'Q_{s}_{a}'
        
        # Current Q value
        Q_current = self.parameters.get(q_key, 0.0)
        
        # Max future Q
        future_keys = [k for k in self.parameters if k.startswith(f'Q_{s_next}_')]
        Q_max_future = max([self.parameters[k] for k in future_keys], default=0.0)
        
        # Update
        Q_new = Q_current + self.eta * (r + gamma * Q_max_future - Q_current)
        self.parameters[q_key] = Q_new
